CREATE TABLE Guest (
	EMAIL		VARCHAR2(30)	NOT NULL,
	NAME		VARCHAR2(30),
	ADDRESS		VARCHAR2(30),
	CITY		VARCHAR2(15),
	POSTCODE	VARCHAR2(8),
	TELEPHONE	VARCHAR2(15),
	PASSWORD	VARCHAR2(30)	NOT NULL,
	CONSTRAINT Email_Guest_PK PRIMARY KEY (EMAIL)
);

CREATE TABLE GuestAdmin (
	EMAIL		VARCHAR2(30)	NOT NULL,
	CONSTRAINT Email_GuestAdmin_PK PRIMARY KEY (EMAIL),
	CONSTRAINT Email_GuestAdmin_FK FOREIGN KEY (EMAIL) REFERENCES Guest(EMAIL) 
);

CREATE TABLE GuestLog (
	GLID		NUMBER(10)	NOT NULL,
	EMAIL		VARCHAR2(30)	NOT NULL,
	IPADDRESS	VARCHAR2(15)	NOT NULL,
	LOGDATE		DATE		NOT NULL,
	CONSTRAINT GLID_GuestLog_PK PRIMARY KEY (GLID),
	CONSTRAINT IPADDRESS_Length CHECK (LENGTH(IPADDRESS) BETWEEN 6 AND 16) 
);


CREATE TABLE GuestEntry (
	GEID		NUMBER(10)	NOT NULL,
	EMAIL		VARCHAR2(30)	NOT NULL,
	ENTRY		VARCHAR2(4000)	NOT NULL,
	ENTRYDATE	DATE,
	CONSTRAINT GEID_GuestEntry_PK PRIMARY KEY (GEID)
);
